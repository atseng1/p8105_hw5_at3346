``` r
set.seed(10)

sim_regression = function(n = 30, beta0 = 2, sd = 50, beta1) {
  
  sim_data = tibble(
    x = rnorm(n, mean = 1, sd = 1),
    y = beta0 + beta1 * x + rnorm(n, 0, 1)
  )

  ls_fit = lm(y ~ x, data = sim_data)
  
  tidy(ls_fit)
}

sim_results = 
  rerun(10000, sim_regression(n = 30, beta0 = 2, sd = 50, beta1 = 0)) %>% 
  bind_rows() %>% 
  select(term, estimate, p.value) %>% 
  filter(term == "x")


sim_results_8 = 
  tibble(beta1_values= c(0, 1, 2, 3, 4, 5, 6)) %>% 
  mutate(
  output_lists = map(.x = beta1_values, ~rerun(10000, sim_regression(beta1 = .x))),
  estimate_dfs = map(output_lists, bind_rows)) %>% 
  select(-output_lists) %>% 
  unnest(estimate_dfs) %>% 
  select(beta1_values, term, estimate, p.value) %>% 
  filter(term == "x") %>% 
  count(p.value < 0.05)
```
